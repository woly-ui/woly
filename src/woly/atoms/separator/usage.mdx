import {Playground} from 'dev/playground'
import {Heading, Input, Separator} from 'ui'

`Sepatator` is a line that divides blocks or content.

### Example

<Playground direction="vertical">
  <div style={{ width: '300px' }}>
    <div style={{ margin: '18px 0' }}>
      <Heading size={2}>User List</Heading>
    </div>
    <Separator priority="primary" />
    <div style={{ margin: '18px 0' }}>
      <Input
        type="text"
        name="name"
        placeholder="Search"
        onChange={() => console.info('On input change')}
        priority="primary"
        style={{ margin: '10px 0' }}
      />
    </div>
  </div>
</Playground>

### Kinds

| Name        | Description    |
| ----------- | -------------- |
| `Separator` | Base separator |

### Props

| Name       | Type     | Default       | Description                                |
| ---------- | -------- | ------------- | ------------------------------------------ |
| `priority` | `string` | `'secondary'` | Priority prop to style Separator component |
