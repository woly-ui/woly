### Multiselect examples

Simple multiselect.

```tsx playground
import { Multiselect, Text } from 'woly';
import { IconInfo } from 'static/icons';
import { sortById } from './sort';
import { block } from 'dev/playground'
import { daysOfWeek } from './constants'

export function Example() {

  const [options, setOptions] = React.useState(daysOfWeek);

  const value = options.filter((option) => option.selected);

  const onChange = (event) => {
    event.preventDefault();

    const id = event.currentTarget.id || event.currentTarget.dataset.id;

    if (!id) {
      return;
    }

    const index = options.findIndex((option) => parseInt(option.id, 10) === parseInt(id, 10));

    if (index > -1) {
      const element = options.splice(index, 1)[0];
      element.selected = !element.selected;

      const newValue = sortById([...options, element], "text");

      setOptions(newValue);
    }
  };

  return <Playground>
    <div style={{padding: '300px 0', width: '100%', display: 'flex', flexDirection: 'column'}}>
      <block.L style={{padding: '10px'}}>
        <Multiselect
          onChange={onChange}
          options={options}
          placeholder="Choose an option"
          priority="primary"
          value={value}
          leftIcon={<IconInfo/>}
          rightIcon={<IconInfo/>}
        />
      </block.L>
      </div>
  </Playground>
}
```
