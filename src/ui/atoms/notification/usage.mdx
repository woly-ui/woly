import { Notification} from './index';
import { Button } from 'ui';
import { Playground, block, StateEvent, State } from 'lib/playground';

`Notification` display the notification message globaly.

### Example

<Playground>
  <State initial={false} change={(i) => !i}>
    {(value, change) => (
      <block.S>
        <Button
          onClick={change}
          text={value ? 'hide notification' : 'show notification'}
          variant="primary"
        />
        <div style={{ display: value ? 'block' : 'none' }}>
          <Notification message="Your transaction has been completed" variant="primary" />
        </div>
      </block.S>
    )}
  </State>
</Playground>

### Kinds

| Name           | Description                                                       |
| -------------- | ----------------------------------------------------------------- |
| `Notification` | Base notification. Useful for creating a new kind of notificaiton |

### Props

| Name      | Type              | Default       | Description                                  |
| --------- | ----------------- | ------------- | -------------------------------------------- |
| `message` | `React.ReactNode` |               | Message that will be shown in notification   |
| `variant` | `string`          | `'secondary'` | Variant prop to style Notification component |
