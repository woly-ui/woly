import * as buttonIcon from './index';
import { OpenEyeIcon, SearchIcon } from '../../../static/icons/';

`buttonIcon` is a button with an icon

### Example

<div style={{ display: 'flex' }}>
  <div
    style={{
      '--bi-background-color': 'bisque',
      '--bi-rounding': '3px',
      margin: '5px',
    }}
  >
    <buttonIcon.Base
      icon={<SearchIcon />}
      onClick={(event) => console.info('buttonIcon clicked')}
    />
  </div>
  <div
    style={{
      '--bi-background-color': '#ffeb3b',
      '--bi-rounding': '3px',
      margin: '5px',
    }}
  >
    <buttonIcon.Base
      icon={<OpenEyeIcon />}
      onClick={(event) => console.info('buttonIcon clicked')}
    />
  </div>
</div>

```tsx
<buttonIcon.Base
  icon={<Icon />}
  onClick={(event) => console.info('buttonIcon clicked')}
/>
```

### Kinds

| Name   | Description              |
| ------ | ------------------------ |
| `Base` | Base button with an icon |

### Props

| Name      | Type                                         | Default | Description                                        |
| --------- | -------------------------------------------- | ------- | -------------------------------------------------- |
| `icon`    | `React.ReactNode`                            |         | Set the icon component inside the button           |
| `onClick` | `React.MouseEventHandler<HTMLButtonElement>` |         | Callback when button clicked                       |
| `...`     | `HTMLButtonElementProps`                     | `{}`    | Other props are inherited from `HTMLButtonElement` |
