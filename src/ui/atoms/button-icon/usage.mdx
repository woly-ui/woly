import { ButtonIcon } from 'ui';
import { Global } from 'box-styles';
import { OpenedEyeIcon, SearchIcon } from 'icons';

`ButtonIcon` is a button with an icon

### Example

<Global>
  <div
    style={{ display: 'flex', width: ' 60px', justifyContent: 'space-between' }}
  >
    <ButtonIcon
      icon={<SearchIcon />}
      onClick={(event) => console.info('ButtonIcon clicked')}
      variant="primary"
    />
    <ButtonIcon
      icon={<OpenedEyeIcon />}
      onClick={(event) => console.info('ButtonIcon clicked')}
      variant="secondary"
    />
  </div>
</Global>

```tsx
<ButtonIcon
  icon={<Icon />}
  onClick={(event) => console.info('ButtonIcon clicked')}
  variant="primary"
/>
```

### Kinds

| Name         | Description              |
| ------------ | ------------------------ |
| `ButtonIcon` | Base button with an icon |

### Props

| Name      | Type                                         | Default     | Description                                        |
| --------- | -------------------------------------------- | ----------- | -------------------------------------------------- |
| `icon`    | `React.ReactNode`                            |             | Set the icon component inside the button           |
| `onClick` | `React.MouseEventHandler<HTMLButtonElement>` |             | Callback when button clicked                       |
| `variant` | `string`                                     | `'default'` | Variant prop to style ButtonIcon component         |
| `...`     | `HTMLButtonElement`                          | `{}`        | Other props are inherited from `HTMLButtonElement` |
