import { ButtonFloating } from 'ui';
import { Global, Playground } from 'box-styles';
import { SearchIcon } from 'icons';

`ButtonFloating` is placed at the bottom right side of the screen and contains an icon

### Example

<Playground>
  <ButtonFloating
    icon={<SearchIcon />}
    onClick={(event) => console.info('ButtonFloating clicked')}
    variant="primary"
  />
</Playground>

### Kinds

| Name             | Description                                              |
| ---------------- | -------------------------------------------------------- |
| `ButtonFloating` | Base round button at the bottom right side of the screen |

### Props

| Name      | Type                                         | Default       | Description                                        |
| --------- | -------------------------------------------- | ------------- | -------------------------------------------------- |
| `icon`    | `React.ReactNode`                            |               | Set the icon component inside the button           |
| `onClick` | `React.MouseEventHandler<HTMLButtonElement>` |               | Callback when button clicked                       |
| `variant` | `string`                                     | `'secondary'` | Variant prop to style ButtonFloating component     |
| `...`     | `HTMLButtonElement`                          | `{}`          | Other props are inherited from `HTMLButtonElement` |
